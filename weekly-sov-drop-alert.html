services/monthlyEngagementShareReport.js
/**
 * services/monthlyEngagementShareReport.js
 *
 * Identifies advertisers whose within-industry engagement share
 * dropped > 20% vs previous 30 days. Returns a flat list sorted by drop% desc.
 */

const moment  = require('moment');
const FbPost  = require('models/fbPost.js');
const IgMedia = require('models/igMedia.js');
const Advertisers = require('models/advertisers.js');

const DROP_THRESHOLD = -0.20;

const aggregateEngagements = (Model, dateField, start, end) =>
  Model.aggregate([
    { $match: {
        [dateField]:  { $gte: start, $lte: end },
        advertisers:  { $exists: true, $not: { $size: 0 } },
        engagements:  { $exists: true, $gt: 0 },
        countries:    { $in: ['HK', 'MO'] },
        adType:       { $in: ['paid_post', 'branded_content'] },
    }},
    { $unwind: '$advertisers' },
    { $group: { _id: '$advertisers', engagements: { $sum: '$engagements' } } },
  ]).allowDiskUse(true);

const getEngagementsMap = async (start, end) => {
  const [fbRows, igRows] = await Promise.all([
    aggregateEngagements(FbPost,   'createdAt',     start, end),
    aggregateEngagements(IgMedia, 'created_time', start, end),
  ]);
  const map = {};
  for (const row of [...fbRows, ...igRows]) {
    const id = row._id.toString();
    map[id] = (map[id] || 0) + row.engagements;
  }
  return map;
};

const getEngagementDropAdvertisers = async () => {
  const now = moment.utc();
  const lastStart = now.clone().subtract(1, 'month').startOf('month').toDate();
  const lastEnd   = now.clone().subtract(1, 'month').endOf('month').toDate();
  const prevStart = now.clone().subtract(2, 'months').startOf('month').toDate();
  const prevEnd   = now.clone().subtract(2, 'months').endOf('month').toDate();

  const lastMonthLabel = now.clone().subtract(1, 'month').format('YYYY-MM');
  const prevMonthLabel = now.clone().subtract(2, 'months').format('YYYY-MM');

  const [lastMap, prevMap] = await Promise.all([
    getEngagementsMap(lastStart, lastEnd),
    getEngagementsMap(prevStart, prevEnd),
  ]);

  // Enrich all advertisers to get their productCategory
  const allAdvIds = [...new Set([...Object.keys(lastMap), ...Object.keys(prevMap)])];
  const advDocs = await Advertisers.find({ _id: { $in: allAdvIds } })
    .select('name fullname logo productCategory regionalInfo')
    .populate('productCategory', 'name')
    .lean();

  const metaMap = {};
  for (const doc of advDocs) metaMap[doc._id.toString()] = doc;

  // Compute industry totals (sum of all advertiser engagements per category)
  const industryLastTotal = {}, industryPrevTotal = {};
  for (const [advId, eng] of Object.entries(lastMap)) {
    const cat = metaMap[advId]?.productCategory?.name;
    if (cat) industryLastTotal[cat] = (industryLastTotal[cat] || 0) + eng;
  }
  for (const [advId, eng] of Object.entries(prevMap)) {
    const cat = metaMap[advId]?.productCategory?.name;
    if (cat) industryPrevTotal[cat] = (industryPrevTotal[cat] || 0) + eng;
  }

  // Compute per-advertiser share drop; filter and collect
  const result = [];
  for (const [advId, lastEng] of Object.entries(lastMap)) {
    const prevEng = prevMap[advId] || 0;
    if (prevEng === 0) continue;

    const meta = metaMap[advId] || {};
    const cat  = meta.productCategory?.name;
    if (!cat) continue;

    const lastShare = lastEng / (industryLastTotal[cat] || 1);
    const prevShare = prevEng / (industryPrevTotal[cat] || 1);
    const dropRate  = (lastShare - prevShare) / prevShare;
    if (dropRate >= DROP_THRESHOLD) continue;

    const regional = meta.regionalInfo?.HK || {};
    result.push({
      advertiserId:  advId,
      categoryName:  cat,
      name:          regional.name || meta.name || meta.fullname || 'Unknown',
      logo:          regional.logo || meta.logo || null,
      prevShare,
      lastShare,
      dropRate,
      prevEngagements: prevEng,
      lastEngagements: lastEng,
      lastMonthLabel,
      prevMonthLabel,
    });
  }

  // Flat sort: most negative dropRate first
  result.sort((a, b) => a.dropRate - b.dropRate);
  return result;
};

module.exports = { getEngagementDropAdvertisers };
