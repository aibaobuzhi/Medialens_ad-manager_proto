// services/monthlySpendingGrowthReport.js — 最終 $match（已實作）
const aggregateSpending = (Model, dateField, start, end) =>
  Model.aggregate([
    { $match: {
        [dateField]: { $gte: start, $lte: end },
        advertisers: { $exists: true, $not: { $size: 0 } },
        spending:    { $exists: true, $gt: 0 },
        countries:   { $in: ['HK', 'MO'] },
        adType:      { $in: ['paid_post', 'branded_content'] },
    }},
    { $unwind: '$advertisers' },
    { $group: { _id: '$advertisers', spending: { $sum: '$spending' } } },
  ]).allowDiskUse(true);
