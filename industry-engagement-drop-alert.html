/**
 * services/monthlyEngagementShareReport.js
 *
 * Identifies advertisers whose within-industry engagement share
 * dropped >= 20% MoM (lastMonth vs prevMonth).
 *
 * Returns results grouped by industry, sorted by drop% desc.
 */

const moment = require('moment');
const FbPost = require('models/fbPost.js');
const IgMedia = require('models/igMedia.js');
const Advertisers = require('models/advertisers.js');

const DROP_THRESHOLD = -0.20; // -20% MoM share drop

// Aggregate engagements per advertiser within a date range
const aggregateEngagements = (Model, dateField, start, end) =>
  Model.aggregate([
    { $match: {
        [dateField]: { $gte: start, $lte: end },
        advertisers: { $exists: true, $not: { $size: 0 } },
        engagements: { $exists: true, $gt: 0 },
        countries:   { $in: ['HK', 'MO'] },
        adType:      { $in: ['paid_post', 'branded_content'] },
    }},
    { $unwind: '$advertisers' },
    { $group: { _id: '$advertisers', engagements: { $sum: '$engagements' } } },
  ]).allowDiskUse(true);

// Returns map: advId -> total engagements
const getMonthlyEngagementsMap = async (start, end) => {
  const [fbRows, igRows] = await Promise.all([
    aggregateEngagements(FbPost,   'createdAt',     start, end),
    aggregateEngagements(IgMedia, 'created_time', start, end),
  ]);
  const map = {};
  for (const row of [...fbRows, ...igRows]) {
    const id = row._id.toString();
    map[id] = (map[id] || 0) + row.engagements;
  }
  return map;
};

const getEngagementDropAdvertisers = async () => {
  const now = moment.utc();
  const lastMonthStart = now.clone().subtract(1, 'month').startOf('month').toDate();
  const lastMonthEnd   = now.clone().subtract(1, 'month').endOf('month').toDate();
  const prevMonthStart = now.clone().subtract(2, 'months').startOf('month').toDate();
  const prevMonthEnd   = now.clone().subtract(2, 'months').endOf('month').toDate();

  const [lastEngMap, prevEngMap] = await Promise.all([
    getMonthlyEngagementsMap(lastMonthStart, lastMonthEnd),
    getMonthlyEngagementsMap(prevMonthStart, prevMonthEnd),
  ]);

  // Enrich with advertiser metadata (productCategory for industry grouping)
  const advIds = Object.keys(lastEngMap);
  const advDocs = await Advertisers.find({ _id: { $in: advIds } })
    .select('name fullname logo productCategory regionalInfo')
    .populate('productCategory', 'name')
    .lean();

  const advMetaMap = {};
  for (const doc of advDocs) advMetaMap[doc._id.toString()] = doc;

  // Group engagements by industry to compute per-industry totals
  const industryLastTotals = {}, industryPrevTotals = {};
  for (const [advId, lastEng] of Object.entries(lastEngMap)) {
    const cat = advMetaMap[advId]?.productCategory?.name;
    if (!cat) continue;
    industryLastTotals[cat] = (industryLastTotals[cat] || 0) + lastEng;
    industryPrevTotals[cat] = (industryPrevTotals[cat] || 0) + (prevEngMap[advId] || 0);
  }

  // Compute per-advertiser share and filter by drop threshold
  const dropList = [];
  for (const [advId, lastEng] of Object.entries(lastEngMap)) {
    const meta = advMetaMap[advId] || {};
    const cat = meta.productCategory?.name;
    if (!cat) continue;

    const prevEng    = prevEngMap[advId] || 0;
    if (prevEng === 0) continue; // No baseline

    const lastShare = lastEng / industryLastTotals[cat];
    const prevShare = prevEng / (industryPrevTotals[cat] || 1);
    const dropRate  = (lastShare - prevShare) / prevShare;

    if (dropRate >= DROP_THRESHOLD) continue; // Not enough drop

    const regional = meta.regionalInfo?.HK || {};
    dropList.push({
      advertiserId: advId,
      industryName: cat,
      name:         regional.name || meta.name || meta.fullname || 'Unknown',
      logo:         regional.logo || meta.logo || null,
      prevShare,
      lastShare,
      dropRate,
      prevEngagements: prevEng,
      lastEngagements: lastEng,
      industryLastEngagements: industryLastTotals[cat],
    });
  }

  // Group by industry, sort advertisers within each group by dropRate asc
  const industryMap = {};
  for (const item of dropList) {
    if (!industryMap[item.industryName]) industryMap[item.industryName] = [];
    industryMap[item.industryName].push(item);
  }
  for (const cat of Object.keys(industryMap)) {
    industryMap[cat].sort((a, b) => a.dropRate - b.dropRate); // most negative first
  }

  // Sort industries — ❓ Option A or B: see open question above
  // Currently using Option B: avg drop of qualifying advertisers in industry
  const industries = Object.entries(industryMap).map(([name, advertisers]) => ({
    name,
    advertisers,
    avgDrop: advertisers.reduce((s, a) => s + a.dropRate, 0) / advertisers.length,
  })).sort((a, b) => a.avgDrop - b.avgDrop); // most negative first

  return industries;
};

module.exports = { getEngagementDropAdvertisers };
